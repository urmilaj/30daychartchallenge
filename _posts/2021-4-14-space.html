  • Get all the unique names of the source and target, where the source is the venture capital investors and the target is the space tech companies.
  • Create the matrix function, which will help put the data in an appropriate format to make the chord diagram.
  • Use d3.chordDirected to which the matrix function will be called creating chord diagram data format.
  • d3.arc will help draw the bands with labels of the chord diagram.
  • d3.ribbonArrow helps draw the links from the source to the target.
  • Finally d3.scaleOrdinal makes the colour scale to colour the venture capital investors.
//Get the unique names from the dataset
const names = Array.from(new Set(data.flatMap(d => [d.source, d.target])));

//Create a function which will make a matrix of the data
const m = () =>  {
const index = new Map(names.map((name, i) => [name, i]));
const matrix = Array.from(index, () => new Array(names.length).fill(0));
for (const {source, target, value} of data) matrix[index.get(source)][index.get(target)]=value
return matrix;
}

//Create variables for the chord diagram
const innerRadius = Math.min(width, height)*0.5,
outerRadius = innerRadius,
chord = d3.chordDirected()
    .padAngle(0.02)
    .sortSubgroups(d3.descending)
    .sortChords(d3.descending),
arc = d3.arc()
    .innerRadius(innerRadius+10)
    .outerRadius(outerRadius),
ribbon = d3.ribbonArrow()
    .headRadius(30)
    .radius(innerRadius)
    .padAngle(10 / innerRadius),
color = d3.scaleOrdinal()
    .domain(["Space Angels","Hempisphere Ventures","DCVC", "RRE Ventures", "Lux Capital", "Marcbell", "Founders Fund"])
    .range(["#ea926c","pink","red","orange","cyan","green","rebeccapurple"])

//Call the matrix data function inside d3.chord to create data the diagram
const chords = chord(m());

svg.append("g")
    .attr("fill-opacity", 0.75)
    .selectAll("g")
    .data(chords)
    .join("path")
    .attr("d", ribbon)
    .attr("class","ribbon")
    .attr("fill", d => color(d.source.index))
    .style("mix-blend-mode", "multiply")

svg.append("g")
    .attr("font-family", "sans-serif")
    .attr("font-size", 10)
    .selectAll("g")
    .data(chords.groups)
    .join("g")
    .call(g => g.append("path") 
    .attr("class","band")
    .attr("id",d => `${names[d.index]}`)
    .attr("d", arc)
    .attr("fill", d => color(d.index),color.unknown("grey"))
    .attr("stroke", d => color(d.index)))
    .call(g => g.append("text")
    .attr("dy", -10)
    .append("textPath")
    .attr("xlink:href", d => `#${names[d.index]}`)
    .attr("startOffset", "22%")
    .text(d => `${names[d.index]}`)
    .style("font-weight","bold")
    .style("text-anchor","middle")
    .style("font-size","1.05em"))
