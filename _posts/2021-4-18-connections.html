# Quentin Tarantino movies and profanity.

# Week 3: Relationships. Day 18: Connections.


Day 18's topic was connections. Five Thirty Eight has a dataset of every profanity used in Quentin Tarantino movies. The data is available for seven movies, with profanity count for each movie. I visualized this dataset as a force directed graph for each movie. Here's what the final visualization looked like below. Each profanity in the movie is represented by a node, nodes that appear bigger in size compared to others indicate a higher frequency of it's use throught the movie compared to other profanities. I was eventually going to put up the visualization on social media and did not know what age groups would come across the visualization and it looked so jarring to me personally, so I decided to use asterisk to edit/blur the words.

Inside the chart function

  1. Group the data based on the word column using d3.group. (line 2)
  2. Using d3.hierarchy create a hierarchial structure of the grouped dataset. (line 4)
  3. The links variable will create the link lines to the nodes. (line 6)
  4. The nodes variable will create the node bubbles/circles. (line 8)
  5. Using d3.scaleSqrt to size the nodes. (line 10)
  6. Using d3.forceSimulation method to create the forced directed graph. (line 14)
  7. An svg variable to append a separate svg for each force-directed graph movie dataset. (line 20)

 

 

 

 

 



 





 






chart = (data,color,title) => {
    const group = d3.group(data,d=>d.word)

    const root = d3.hierarchy(group)

    const links = root.links()

    const nodes = root.descendants()

    const radius = d3.scaleSqrt()
        .domain(d3.extent(nodes.filter(d=>d.depth==2),d=>d.data.count))
        .range([15,50])

    const simulation = d3.forceSimulation(nodes.filter(d=>d.depth==2))
        .force("link",d3.forceLink(links).distance(0).strength(0.02))
        .force("charge",d3.forceManyBody().strength(-100))
        .force("x",d3.forceX().strength(0.05))
        .force("y",d3.forceY().strength(0.05))
          
    const svg = d3.select(".chart")
        .append("svg")
        .attr("width",width+margin.left+margin.right)
        .attr("height",height+margin.top+margin.bottom)
        .attr("viewBox", [-width / 2, -height / 2, width, height-50])
        .style("background-color","black")

svg.append("text")
    .attr("y",-250)
    .attr("text-anchor","middle")
    .style("font-size","3em")
    .style("font-family","Showcard Gothic")
    .style("fill","white")
    .text(title)

const link = svg.append("g")
    .attr("stroke",color)
    .attr("stroke-opacity",1)
    .attr("stroke-width",2)
    .selectAll("line")
    .data(links)
    .join("line")

const node = svg.append("g")
    .attr("fill",color)
    .attr("stroke","black")
    .selectAll("circle")
    .data(nodes)
    .join("circle")
    .attr("r",d=>d.depth==2?radius(d.data.count):5)


const text = svg.append("g")
    .attr("class", "text")
    .selectAll("text")
    .data(nodes)
    .join("text")
    .text(d => d.depth==2?d.data.word+": "+d.data.count:" ")
    .style("fill","white")
        
            

simulation.on("tick", ()=>{
    link.attr("x1",d=>d.source.x)
    .attr("y1",d=>d.source.y)
    .attr("x2",d=>d.target.x)
    .attr("y2",d=>d.target.y)
    
    node.attr("cx",d=>d.x)
    .attr("cy",d=>d.y)

    text.attr("x",d=>d.x)
    .attr("y",d=>d.y)
})
            
}
