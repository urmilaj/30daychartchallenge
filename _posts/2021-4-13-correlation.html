Create the scales and the axis for the scatterplot. Usine simple statistics library to draw the regression line.

const tempScale = d3.scaleLinear()
    .domain(d3.extent(data,d=>d.Temperature))
    .range([margin.top,width-margin.top])

const xAxis = g => g.attr("transform",`translate(0,${height-margin.left})`)
    .call(d3.axisBottom(tempScale))

const y = d3.scaleLinear()
    .domain(d3.extent(data,d=>d.RentedBikeCount))
    .range([height-margin.left,margin.top])

const yAxis = g => g.attr("transform",`translate(${margin.top},0)`)
    .call(d3.axisLeft(y))
            
const lR = ss.linearRegression(data.map(d=>[d.Temperature,d.RentedBikeCount]))
const line = ss.linearRegressionLine(lR)   

const regressionLine = d3.line()
    .x(d=>tempScale(d.x))
    .y(d=>y(d.y))

point = () => {
    const firstX = -12
    const lastX = 39.4
    const xCoordinates = [firstX, lastX]

    return xCoordinates.map(d=>({
    x: d,
    y: line(d)
}))}

svg.append("text")
    .attr("x",width/2)
    .attr("y",0)
    .style("text-anchor","middle")
    .text("Seoul Bike Sharing")
    .style("fill","#d96d3c")
    .style("font-size","2em")
    .style("font-weight","bold")

svg.append("g")
    .call(xAxis)
    .style("color","#d96d3c")
    .style("font-size","0.9em")
    .style("font-weight","bold")

svg.append("g")
    .call(yAxis)
    .style("color","#d96d3c")
    .style("font-size","0.8em")
    .style("font-weight","bold")

svg.append("g")
    .selectAll("circle")
    .data(data)
    .join("circle")
    .attr("cx",d=>tempScale(d.Temperature))
    .attr("cy",d=>y(d.RentedBikeCount))
    .attr("r",1.2)
    .style("fill","#d96d3c")

svg.append("path")
    .datum(point)
    .attr("d",regressionLine)
    .attr("fill","none")
    .attr("stroke","white")
    .style("stroke-width",2.5)

svg.append("text")
    .attr("x",-height/2+80)
    .attr("y",-5)
    .text("Number of bikes rented")
    .style("text-anchor","end")
    .attr("transform","rotate(-90)")
    .style("fill","#d96d3c")
    .style("font-weight","bold")

svg.append("text")
    .attr("x",width/2)
    .attr("y",height)
    .style("text-anchor","middle")
    .text("Temperature in °C")
    .style("fill","#d96d3c")
    .style("font-weight","bold")
